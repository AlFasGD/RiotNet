760  private static string CreateDataParam(IEnumerable<string> propertyNames, Type type, Type listType = null)
761  {
762  return string.Join(",", propertyNames.Select(p => CorrectPropertyNameCase(p, type, listType)).Where(p => p != null));
763  }
764 
765  private static string CorrectPropertyNameCase(string propertyName, Type type, Type listType = null)
766  {
767  if (string.IsNullOrEmpty(propertyName))
768  return null;
769  if (string.Equals(propertyName, "all", StringComparison.InvariantCultureIgnoreCase))
770  return "all";
771 
772  var property = type.GetProperty(propertyName, BindingFlags.IgnoreCase | BindingFlags.Public | BindingFlags.Instance | BindingFlags.FlattenHierarchy);
773  if (property == null && listType != null)
774  property = listType.GetProperty(propertyName, BindingFlags.IgnoreCase | BindingFlags.Public | BindingFlags.Instance | BindingFlags.FlattenHierarchy);
775  if (property != null)
776  {
777  // If there is a JsonProperty attribute, get the name from the attribute.
778  var jsonPropertyAttribute = property.GetCustomAttribute<JsonPropertyAttribute>();
779  if (jsonPropertyAttribute != null && !string.IsNullOrEmpty(jsonPropertyAttribute.PropertyName))
780  return jsonPropertyAttribute.PropertyName;
781  }
782  return propertyName.Remove(1).ToLowerInvariant() + propertyName.Substring(1);
783  }
