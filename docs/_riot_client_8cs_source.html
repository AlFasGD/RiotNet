293  protected virtual T Execute<T>(IRestRequest request, IRestClient client) where T : new()
294  {
295  var attemptCount = 0;
296  do
297  {
298  var response = client.Execute<T>(request);
299  ++attemptCount;
300  var action = DetermineResponseAction(response, attemptCount);
301  if (action == ResponseAction.Return)
302  return response.Data;
303  if (action == ResponseAction.ReturnDefault)
304  break;
305  if (action == ResponseAction.Retry)
306  {
307  var retryAfterHeader = response.Headers.FirstOrDefault(h => string.Equals(h.Name, "Retry-After", StringComparison.InvariantCultureIgnoreCase));
308  if (retryAfterHeader != null)
309  {
310  int delaySeconds;
311  if (int.TryParse(retryAfterHeader.Value as string, out delaySeconds))
312  Thread.Sleep((delaySeconds + 1) * 1000);
313  }
314  }
315  } while (attemptCount < Settings.MaxRequestAttempts);
316 
317  return default(T);
318  }

371  protected virtual ResponseAction DetermineResponseAction(IRestResponse response, int attemptCount)
372  {
373  if (response.ResponseStatus == ResponseStatus.TimedOut)
374  {
375  var args = new RetryEventArgs(response, attemptCount);
376  args.Retry = Settings.RetryOnTimeout;
377  OnRequestTimedOut(args);
378  if (args.Retry)
379  return ResponseAction.Retry;
380  if (Settings.ThrowOnError)
381  throw new RestTimeoutException(response);
382 
383  return ResponseAction.ReturnDefault;
384  }
385  if (response.ResponseStatus == ResponseStatus.Error && response.StatusCode == 0)
386  {
387  var args = new RetryEventArgs(response, attemptCount);
388  args.Retry = Settings.RetryOnConnectionFailure;
389  OnConnectionFailed(args);
390  if (args.Retry)
391  return ResponseAction.Retry;
392  if (Settings.ThrowOnError)
393  throw new ConnectionFailedException(response, response.ErrorException);
394 
395  return ResponseAction.ReturnDefault;
396  }
397  var statusCode = (int)response.StatusCode;
398  if (statusCode == 429)
399  {
400  var args = new RetryEventArgs(response, attemptCount);
401  args.Retry = Settings.RetryOnRateLimitExceeded;
402  OnRateLimitExceeded(args);
403  if (args.Retry)
404  return ResponseAction.Retry;
405  if (Settings.ThrowOnError)
406  throw new RateLimitExceededException(response);
407 
408  return ResponseAction.ReturnDefault;
409  }
410  if (statusCode == 404)
411  {
412  OnResourceNotFound(new ResponseEventArgs(response));
413  if (Settings.ThrowOnNotFound)
414  throw new NotFoundException(response);
415 
416  return ResponseAction.ReturnDefault;
417  }
418  if (statusCode >= 500)
419  {
420  var args = new RetryEventArgs(response, attemptCount);
421  args.Retry = Settings.RetryOnServerError;
422  OnServerError(args);
423  if (args.Retry)
424  return ResponseAction.Retry;
425  if (Settings.ThrowOnError)
426  throw new RestException(response);
427 
428  return ResponseAction.ReturnDefault;
429  }
430  if (statusCode >= 400 || response.ResponseStatus != ResponseStatus.Completed)
431  {
432  OnResponseError(new ResponseEventArgs(response));
433  if (Settings.ThrowOnError)
434  throw new RestException(response, response.ErrorException);
435 
436  return ResponseAction.ReturnDefault;
437  }
438  return ResponseAction.Return;
439  }
